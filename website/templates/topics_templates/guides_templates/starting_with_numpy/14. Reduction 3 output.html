print_matrix("x", x)
# reduce on all axes
print("max(x)=", np.max(x)) 
# reduce on rows
print_matrix("max_{0}(x)", np.max(x, axis=0))
# reduce on columns
print_matrix("max_{1}(x)", np.max(x, axis=1))
# same as all axes in this case
print("max_{0,1}(x)=", np.max(x, axis=(0,1)))
